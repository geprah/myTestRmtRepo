*****MODELING*****
//Teacher can handle more than one subject
//Teacher can teach more than one student
Teacher{
teacherId
teacherName
}


Subject{
subjectId
subjectName
teacherId_fk
primary key(subjectId,teacherId)
}

Class{
classId 
classSection
subjectId_fk
primary key(classId)
}

//Student can study more than one subject
//Student can have only one class(therefore one teacher)
Students{
studentId
studentName
classId_fk
}

*****END OF MODELING*****

*****IMPLEMENTING DB*****
create database ladb;
use ladb;

create table Teacher(teacherId int auto_increment, 
teacherName varchar(20), primary key(teacherId));

create table Subject(subjectId int auto_increment,
subjectName varchar(20), teacherId int,
foreign key (teacherId) references Teacher(teacherId),
primary key(subjectId,teacherId));

create table Class(classId int auto_increment,
classSection varchar(20),subjectId int,
foreign key(subjectId) references Subject(subjectId),
primary key(classId,subjectId));

create table Student(studentId int auto_increment,
studentName varchar(20), classId int, 
foreign key (classId) references Class(classId), 
primary key(studentId));
